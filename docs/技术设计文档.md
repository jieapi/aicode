# AI 代码编辑器 - 安卓技术设计文档

## 1. 项目概述

安卓原生 AI 代码编辑器，支持语法高亮、智能代码补全、多提供商 AI 服务管理、文件树管理。

**目标平台**: Android 16 (API 36)

## 2. 技术栈

| 层级 | 技术 | 版本 |

| 依赖注入 | Hilt | 2.51+ |
| 架构模式 | MVVM + Clean Architecture | - |

```
├── 括号匹配
├── 多光标编辑
└── 主题切换
```

**WebView 桥接:**
```kotlin
@Composable
fun CodeEditorWebView(
    code: String,
    language: String,
    theme: String,
    onCodeChange: (String) -> Unit,
    onCursorChange: (Int, Int) -> Unit,
    modifier: Modifier = Modifier
) {
    AndroidView(
        factory = { context ->
            WebView(context).apply {
                settings.javaScriptEnabled = true
                settings.domStorageEnabled = true
                webViewClient = CodeEditorWebViewClient()
                addJavascriptInterface(
                    CodeEditorBridge(onCodeChange, onCursorChange),
                    "AndroidBridge"
                )
                loadUrl("file:///android_asset/editor/index.html")
            }
        },
        update = { webView ->
            webView.evaluateJavascript(
                "updateCode('${code.escapeJs()}')", null
            )
            webView.evaluateJavascript(
                "setLanguage('$language')", null
            )
            webView.evaluateJavascript(
                "setTheme('$theme')", null
            )
        },
        modifier = modifier.fillMaxSize()
    )
}

class CodeEditorBridge(
    private val onCodeChange: (String) -> Unit,
    private val onCursorChange: (Int, Int) -> Unit
) {
    @JavascriptInterface
    fun onCodeChanged(code: String) {
        onCodeChange(code)
    }

    @JavascriptInterface
    fun onCursorChanged(line: Int, column: Int) {
        onCursorChange(line, column)
    }
}
```

```kotlin
@Composable
fun FileTreeView(
    root: FileNode,
    onFileClick: (FileNode) -> Unit,
    onDirectoryToggle: (FileNode) -> Unit,
    modifier: Modifier = Modifier
) {
    LazyColumn(modifier = modifier) {
        fun renderNode(node: FileNode) {
            item(key = node.path) {
                FileTreeItem(
                    node = node,
                    onClick = {
                        if (node.isDirectory) {
                            onDirectoryToggle(node)
                        } else {
                            onFileClick(node)
                        }
                    }
                )
            }
            if (node.isExpanded) {
                node.children.forEach { child ->
                    renderNode(child)
                }
            }
        }
        renderNode(root)
    }
}

@Composable
fun FileTreeItem(
    node: FileNode,
    onClick: () -> Unit
) {
    Row(
        modifier = Modifier
            .fillMaxWidth()
            .padding(start = (node.level * 16).dp)
            .clickable(onClick = onClick)
            .padding(vertical = 4.dp, horizontal = 8.dp),
        verticalAlignment = Alignment.CenterVertically
    ) {
        if (node.isDirectory) {
            Icon(
                imageVector = if (node.isExpanded) {
                    Icons.Default.KeyboardArrowDown
                } else {
                    Icons.Default.KeyboardArrowRight
                },
                contentDescription = null,
                modifier = Modifier.size(16.dp)
            )
        } else {
            Spacer(modifier = Modifier.width(16.dp))
        }

        Spacer(modifier = Modifier.width(4.dp))

        Icon(
            imageVector = getFileIcon(node),
            contentDescription = null,
            modifier = Modifier.size(16.dp),
            tint = getFileIconColor(node)
        )

        Spacer(modifier = Modifier.width(8.dp))

        Text(
            text = node.name,
            style = MaterialTheme.typography.bodyMedium
        )
    }
}

fun getFileIcon(node: FileNode): ImageVector {
    if (node.isDirectory) {
        return if (node.isExpanded) Icons.Default.FolderOpen else Icons.Default.Folder
    }
    return when (node.name.substringAfterLast('.')) {
        "kt", "kotlin" -> Icons.Default.Code
        "java" -> Icons.Default.Code
        "xml" -> Icons.Default.Code
        "json" -> Icons.Default.DataObject
        "md" -> Icons.Default.Description
        "txt" -> Icons.Default.TextSnippet
        else -> Icons.Default.InsertDriveFile
    }
}

fun getFileIconColor(node: FileNode): Color {
    if (node.isDirectory) return Color(0xFF4FC3F7)
    return when (node.name.substringAfterLast('.')) {
        "kt", "kotlin" -> Color(0xFF7C4DFF)
        "java" -> Color(0xFFFF7043)
        "xml" -> Color(0xFF66BB6A)
        "json" -> Color(0xFFFFCA28)
        "md" -> Color(0xFF42A5F5)
        else -> Color.Gray
    }
}
```

**文件操作:**
```kotlin
class FileSystemRepository @Inject constructor(
    @ApplicationContext private val context: Context
) {
    suspend fun getRootDirectory(): FileNode = withContext(Dispatchers.IO) {
        val root = context.getExternalFilesDir(null) ?: context.filesDir
        buildFileTree(root, level = 0)
    }

    private fun buildFileTree(file: File, level: Int): FileNode {
        return FileNode(
            name = file.name,
            path = file.absolutePath,
            isDirectory = file.isDirectory,
            level = level,
            children = if (file.isDirectory) {
                file.listFiles()
                    ?.sortedWith(compareBy<File> { !it.isDirectory }.thenBy { it.name })
                    ?.map { buildFileTree(it, level + 1) }
                    ?: emptyList()
            } else {
                emptyList()
            }
        )
    }

    suspend fun readFile(path: String): String = withContext(Dispatchers.IO) {
        File(path).readText()
    }

    suspend fun writeFile(path: String, content: String) = withContext(Dispatchers.IO) {
        File(path).writeText(content)
    }

    suspend fun createFile(parentPath: String, name: String): File = withContext(Dispatchers.IO) {
        val file = File(parentPath, name)
        file.createNewFile()
        file
    }

    suspend fun createDirectory(parentPath: String, name: String): File = withContext(Dispatchers.IO) {
        val dir = File(parentPath, name)
        dir.mkdirs()
        dir
    }

    suspend fun deleteFile(path: String): Boolean = withContext(Dispatchers.IO) {
        File(path).deleteRecursively()
    }

    suspend fun renameFile(oldPath: String, newName: String): File = withContext(Dispatchers.IO) {
        val oldFile = File(oldPath)
        val newFile = File(oldFile.parent, newName)
        oldFile.renameTo(newFile)
        newFile
    }
}
```

**数据模型:**
```kotlin

data class AIProvider(
    @PrimaryKey val id: String,
    val name: String,
    val type: ProviderType,
    val apiKey: String,
    val baseUrl: String,
    val models: List<String>,
    val defaultModel: String,
    val isActive: Boolean,
    val createdAt: Long,
    val temperature: Float = 0.2f,
    val maxTokens: Int = 256,
    val topP: Float = 1.0f

enum class ProviderType(val displayName: String) {
    OPENAI("OpenAI"),
    ANTHROPIC("Anthropic"),
    GOOGLE("Google AI"),
    DEEPSEEK("DeepSeek"),
    OLLAMA("Ollama"),
    CUSTOM("自定义")

```

**适配器模式:**
```kotlin
interface AIProviderAdapter {
    suspend fun complete(request: CompletionRequest): AIResponse
    fun completeStream(request: CompletionRequest): Flow<String>
    suspend fun listModels(): List<String>
    suspend fun validateConnection(): Boolean
}

class ProviderRegistry @Inject constructor() {
    private val factories = mutableMapOf<ProviderType, (AIProvider) -> AIProviderAdapter>()

    fun register(type: ProviderType, factory: (AIProvider) -> AIProviderAdapter) {
        factories[type] = factory
    }

    fun getAdapter(provider: AIProvider): AIProviderAdapter {
        val factory = factories[provider.type]
            ?: throw IllegalArgumentException("不支持的提供商: ${provider.type}")
        return factory(provider)
    }
}
```

## 9. 安全设计

- API Key 使用 AndroidKeyStore + EncryptedSharedPreferences 加密存储
- 所有 AI 请求仅通过 HTTPS
- 用户可审查发送给 AI 的具体内容

```kotlin
class StandardAgentWorkflow @Inject constructor(
    private val providerRegistry: ProviderRegistry,
    private val toolRegistry: ToolRegistry,
    private val codeChangeTracker: CodeChangeTracker
) : AgentWorkflow {

    override suspend fun execute(
        userRequest: String,
        context: AgentContext,
        tools: List<AgentTool>
    ): WorkflowResult {
        val messages = mutableListOf<AgentMessage>()
        val changes = mutableListOf<CodeChange>()
        val errors = mutableListOf<String>()

        try {
            messages.add(AgentMessage.UserMessage(userRequest))

            // 构建 system prompt
            val systemPrompt = buildSystemPrompt(context, tools)
            val provider = providerRegistry.getActiveProvider()
            val adapter = providerRegistry.getAdapter(provider)

            // 主循环：重复调用 AI 和工具，直到完成
            var maxIterations = 10
            var isComplete = false

            while (!isComplete && maxIterations > 0) {
                maxIterations--

                // 调用 AI 获取响应或工具调用
                val aiResponse = adapter.complete(
                    CompletionRequest(
                        messages = messages.toChatMessages(systemPrompt),
                        temperature = 0.2f,
                        maxTokens = 2048,
                        tools = tools.map { it.toToolDefinition() }
                    )
                )

                if (aiResponse.toolCalls.isNotEmpty()) {
                    // 执行工具调用
                    val assistantMsg = AgentMessage.AssistantMessage(
                        content = aiResponse.content,
                        toolCalls = aiResponse.toolCalls
                    )
                    messages.add(assistantMsg)

                    for (toolCall in aiResponse.toolCalls) {
                        try {
                            val tool = toolRegistry.getTool(toolCall.name)
                            val result = tool.execute(toolCall.arguments)

                            // 追踪代码变更
                            if (toolCall.name in listOf("write_file", "replace_code", "insert_code")) {
                                changes.addAll(codeChangeTracker.trackChange(toolCall, result))
                            }

                            messages.add(AgentMessage.ToolResultMessage(toolCall.name, result))
                        } catch (e: Exception) {
                            errors.add("工具执行失败: ${toolCall.name} - ${e.message}")
                            messages.add(
                                AgentMessage.ToolResultMessage(
                                    toolCall.name,
                                    ToolResult.Error(e.message ?: "未知错误")
                                )
                            )
                        }
                    }
                } else {
                    // AI 返回最终响应
                    messages.add(AgentMessage.AssistantMessage(content = aiResponse.content))
                    isComplete = true
                }
            }

            return WorkflowResult(
                status = if (errors.isEmpty()) WorkflowStatus.SUCCESS else WorkflowStatus.PARTIAL_SUCCESS,
                result = messages.filterIsInstance<AgentMessage.AssistantMessage>().lastOrNull()?.content ?: "",
                changes = changes,
                errors = errors,
                messages = messages
            )

        } catch (e: Exception) {
            return WorkflowResult(
                status = WorkflowStatus.FAILED,
                result = "工作流执行失败",
                errors = listOf(e.message ?: "未知错误"),
                messages = messages
            )
        }
    }

    private fun buildSystemPrompt(context: AgentContext, tools: List<AgentTool>): String {
        val toolDescriptions = tools.joinToString("\n") { tool ->
            """
            工具: ${tool.name}
            描述: ${tool.description}
            参数: ${tool.parameters.values.joinToString(", ") { "${it.name}(${it.type})" }}
            """.trimIndent()
        }

        return """
            你是一个 AI 编程助手，具有以下能力：
            - 阅读和分析代码
            - 根据用户需求修改代码
            - 创建新文件和函数
            - 检查代码质量
            
            项目信息:
            - 根目录: ${context.projectRoot}
            - 当前文件: ${context.currentFile}
            - 语言: ${context.language}
            
            可用工具:
            $toolDescriptions
            
            操作规则:
            1. 使用工具前，先理解用户需求
            2. 一次工具调用完成一个明确的任务
            3. 检查工具执行结果，如果失败则尝试替代方案
            4. 最后总结所做的更改
            5. 优先使用工具，避免在响应中粘贴完整的代码
        """.trimIndent()
    }
}
```

```kotlin
@HiltViewModel
class AIAgentViewModel @Inject constructor(
    private val agentWorkflow: AgentWorkflow,
    private val toolRegistry: ToolRegistry,
    private val editorRepository: EditorRepository,
    private val codeChangeTracker: CodeChangeTracker
) : ViewModel() {

    private val _agentState = MutableStateFlow<AgentUIState>(AgentUIState.Idle)
    val agentState: StateFlow<AgentUIState> = _agentState.asStateFlow()

    private val _messages = MutableStateFlow<List<AgentUIMessage>>(emptyList())
    val messages: StateFlow<List<AgentUIMessage>> = _messages.asStateFlow()

    private val _changes = MutableStateFlow<List<CodeChange>>(emptyList())
    val changes: StateFlow<List<CodeChange>> = _changes.asStateFlow()

    fun executeAgentRequest(
        request: String,
        currentFile: String? = null,
        selectedCode: String? = null
    ) = viewModelScope.launch {
        _agentState.value = AgentUIState.Loading

        try {
            val context = AgentContext(
                currentFile = currentFile,
                selectedCode = selectedCode,
                projectRoot = editorRepository.getProjectRoot(),
                language = currentFile?.let { detectLanguage(it) }
            )

            val result = agentWorkflow.execute(
                userRequest = request,
                context = context,
                tools = toolRegistry.getAvailableTools()
            )

            _agentState.value = AgentUIState.Result(result.status)
            _messages.value = result.messages.toUIMessages()
            _changes.value = result.changes

            if (result.status != WorkflowStatus.SUCCESS) {
                _agentState.value = AgentUIState.Error(result.errors.joinToString("\n"))
            }

        } catch (e: Exception) {
            _agentState.value = AgentUIState.Error(e.message ?: "未知错误")
        }
    }

    fun applyChanges(changes: List<CodeChange>) = viewModelScope.launch {
        try {
            for (change in changes) {
                when (change.type) {
                    ChangeType.CREATE -> editorRepository.createFile(change.filePath, change.newCode)
                    ChangeType.REPLACE -> editorRepository.replaceCode(
                        change.filePath,
                        change.startLine,
                        change.endLine,
                        change.newCode
                    )
                    ChangeType.INSERT -> editorRepository.insertCode(
                        change.filePath,
                        change.startLine,
                        0,
                        change.newCode
                    )
                    ChangeType.DELETE -> editorRepository.deleteCodeRange(
                        change.filePath,
                        change.startLine,
                        change.endLine
                    )
                    else -> {} // 其他操作
                }
            }
            _agentState.value = AgentUIState.Applied
        } catch (e: Exception) {
            _agentState.value = AgentUIState.Error("应用更改失败: ${e.message}")
        }
    }

    fun rejectChanges() {
        _agentState.value = AgentUIState.Idle
        _changes.value = emptyList()
    }

    private fun detectLanguage(filePath: String): String {
        return when (filePath.substringAfterLast(".")) {
            "kt", "kotlin" -> "kotlin"
            "java" -> "java"
            "dart" -> "dart"
            "py" -> "python"
            "js" -> "javascript"
            "ts" -> "typescript"
            else -> "unknown"
        }
    }
}
```

```kotlin
fun ChangeItem(change: CodeChange) {
    Row(
        modifier = Modifier
            .fillMaxWidth()
            .padding(4.dp),
        verticalAlignment = Alignment.CenterVertically
    ) {
        Text(
            text = when (change.type) {
                ChangeType.CREATE -> "+"
                ChangeType.DELETE -> "−"
                ChangeType.REPLACE -> "~"
                else -> "→"
            },
            modifier = Modifier.padding(end = 8.dp),
            color = when (change.type) {
                ChangeType.CREATE -> Color.Green
                ChangeType.DELETE -> Color.Red
                else -> Color.Yellow
            }
        )
        Text(
            text = "${change.filePath} (L${change.startLine}-${change.endLine})",
            style = MaterialTheme.typography.labelSmall,
            modifier = Modifier.weight(1f)
        )
    }
}
```

### 10.6 工具注册和依赖注入

```kotlin
@Module
@InstallIn(SingletonComponent::class)
object AgentModule {

    @Provides
    @Singleton
    fun provideToolRegistry(
        readFileTool: ReadFileTool,
        writeFileTool: WriteFileTool,
        listFilesTool: ListFilesTool,
        createFileTool: CreateFileTool,
        parseCodeTool: ParseCodeTool,
        lintCheckTool: LintCheckTool,
        insertCodeTool: InsertCodeTool,
        replaceCodeTool: ReplaceCodeTool,
        formatCodeTool: FormatCodeTool
    ): ToolRegistry {
        return ToolRegistry().apply {
            register("read_file", readFileTool)
            register("write_file", writeFileTool)
            register("list_files", listFilesTool)
            register("create_file", createFileTool)
            register("parse_code", parseCodeTool)
            register("lint_check", lintCheckTool)
            register("insert_code", insertCodeTool)
            register("replace_code", replaceCodeTool)
            register("format_code", formatCodeTool)
        }
    }

    @Provides
    @Singleton
    fun provideAgentWorkflow(
        providerRegistry: ProviderRegistry,
        toolRegistry: ToolRegistry,
        codeChangeTracker: CodeChangeTracker
    ): AgentWorkflow {
        return StandardAgentWorkflow(providerRegistry, toolRegistry, codeChangeTracker)
    }
}

class ToolRegistry {
    private val tools = mutableMapOf<String, AgentTool>()

    fun register(name: String, tool: AgentTool) {
        tools[name] = tool
    }

    fun getTool(name: String): AgentTool {
        return tools[name] ?: throw IllegalArgumentException("工具不存在: $name")
    }

    fun getAvailableTools(): List<AgentTool> {
        return tools.values.toList()
    }
}
```
